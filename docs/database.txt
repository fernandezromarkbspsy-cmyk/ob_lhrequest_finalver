tables:

clusters:
- cluster_name = Cluster lookup/dropdown value
- hub_name = Hub Name lookup/dropdown detail
- region = Region lookup value
- dock_number = Dock No. lookup value
- backlogs = Backlogs lookup value
- backlogs_ts = Backlogs timestamp

users:
- name = user name
- role = user role (fte_ops, fte_mm, ops_pic, dock_officer, doc_officer)
- ops_id = Backroom login identifier
- email = FTE login identifier
- password_hash = Backroom password/change-password storage
- first_time_login = whether the user still needs to change initial password

requests:
- Stores all linehaul requests
- request_timestamp
- cluster
- region
- dock_no
- backlogs
- backlogs_ts
- ob_fte
- ob_ops_pic
- midmile_fte
- truck_size
- truck_type
- plate_number
- provide_time
- linehaul_trip_no
- docked_time
- status (PENDING, APPROVED, ASSIGNED, FOR_DOCKING, DOCKED, CONFIRMED, CANCELLED, REJECTED_BY_MM)
- rejection_remarks
- approved_at
- rejected_at
- confirmed_at
- created_at
- updated_at

request_events:
- request_id = related request ID
- event_type = workflow event name
- action = action that created the event
- status = request status after the event
- previous_status = request status before the event
- payload = JSON event payload
- created_at

notifications:
- role = recipient role
- request_id = related request ID
- event_type = workflow event name
- message = notification message
- should_play_sound = whether frontend should play the alert sound
- sound_played_at
- read_at
- created_at

user_otps:
- email = FTE email
- code_hash = bcrypt hash of OTP code
- expires_at
- used_at
- created_at
